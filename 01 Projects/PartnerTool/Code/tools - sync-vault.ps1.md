# tools\sync-vault.ps1

```powershell
<#
.SYNOPSIS
  Regenerate the Obsidian vault's PartnerTool code snapshot from this repo.

.DESCRIPTION
  Mirrors every source file into <vault>\01 Projects\PartnerTool\Code\*.md
  (one note per file: source-path frontmatter + fenced code block), refreshes
  _Code Index.md, and rewrites Changelog.md from versions.md.

  Uses .NET file IO (UTF-8, no BOM) on purpose: PS 5.1 Get-Content/Set-Content
  default to the ANSI code page and silently mangle UTF-8 (the vault's dot/dash/
  bullet glyphs). No-ops gracefully when the vault isn't present on this machine
  (e.g. the host uses a different path, or CI) so it is safe to run from a hook.

.PARAMETER VaultPath
  Root of the Obsidian vault. Defaults to $env:PARTNERTOOL_VAULT, else
  'C:\Obsidian Vaults\Vibe Projects'.

.PARAMETER PruneOrphans
  Delete Code\ notes whose source file no longer exists. Off by default (the
  auto hook never prunes); run manually with -PruneOrphans to clean up after a
  source file is renamed or removed.

.EXAMPLE
  powershell -File tools\sync-vault.ps1
.EXAMPLE
  powershell -File tools\sync-vault.ps1 -VaultPath 'D:\Vaults\Vibe Projects' -PruneOrphans
#>
[CmdletBinding()]
param(
  [string]$VaultPath = $(if ($env:PARTNERTOOL_VAULT) { $env:PARTNERTOOL_VAULT } else { 'C:\Obsidian Vaults\Vibe Projects' }),
  [switch]$PruneOrphans
)

$ErrorActionPreference = 'Stop'

$repo = Split-Path $PSScriptRoot -Parent          # tools\ -> repo root
$proj = Join-Path $VaultPath '01 Projects\PartnerTool'
$code = Join-Path $proj 'Code'

if (-not (Test-Path -LiteralPath $VaultPath)) {
  Write-Host "sync-vault: vault not found at '$VaultPath' - skipping (set PARTNERTOOL_VAULT or pass -VaultPath)."
  exit 0
}
if (-not (Test-Path -LiteralPath $code)) { New-Item -ItemType Directory -Force $code | Out-Null }

$enc = New-Object System.Text.UTF8Encoding($false)   # UTF-8, no BOM

function Get-Lang($ext) {
  switch ($ext) {
    '.cs'       { 'csharp' }
    '.xaml'     { 'xml' }
    '.csproj'   { 'xml' }
    '.wxs'      { 'xml' }
    '.manifest' { 'xml' }
    '.md'       { 'markdown' }
    '.ps1'      { 'powershell' }
    default     { '' }
  }
}

# --- collect source files (same scope as the DR snapshot) ---
$files = Get-ChildItem $repo -Recurse -File -Include *.cs,*.xaml,*.csproj,*.wxs,*.manifest,*.ps1 |
  Where-Object { $_.FullName -notmatch '\\obj\\' -and $_.FullName -notmatch '\\bin\\' }
$files += Get-Item (Join-Path $repo 'versions.md')

$expected = @{}
$rows = New-Object System.Collections.Generic.List[string]

foreach ($f in ($files | Sort-Object FullName)) {
  $rel  = $f.FullName.Substring($repo.Length + 1)
  $note = ($rel -replace '\\', ' - ') + '.md'
  $expected[$note] = $true

  $src  = [IO.File]::ReadAllText($f.FullName)
  $open = '```' + (Get-Lang $f.Extension)            # precompute: PS 5.1 comma binds tighter than +
  $body = "---`nproject: PartnerTool`ntags: [partnertool, code]`nsource-path: $rel`n---`n`n# $rel`n`n" +
          $open + "`n" + $src.TrimEnd() + "`n" + '```' + "`n"
  [IO.File]::WriteAllText((Join-Path $code $note), $body, $enc)
  $rows.Add("- [[" + ($note -replace '\.md$', '') + "|" + $rel + "]]")
}

# --- _Code Index.md (live count / date / version) ---
$csproj  = [IO.File]::ReadAllText((Join-Path $repo 'PartnerTool\PartnerTool.csproj'))
$version = [regex]::Match($csproj, '<Version>([^<]+)</Version>').Groups[1].Value
$today   = (Get-Date).ToString('d MMMM yyyy')
$index = "---`nproject: PartnerTool`ntags: [partnertool, code]`n---`n`n# Code Snapshot Index`n`n" +
         "Auto-generated by ``tools\sync-vault.ps1`` on **$today** at **v$version** ($($rows.Count) files).`n" +
         "Live repo (source of truth): GitHub ``N3cTr0/PartnerTool``. Restore procedure: [[Restore From Snapshot]].`n`n" +
         ($rows -join "`n") + "`n"
[IO.File]::WriteAllText((Join-Path $code '_Code Index.md'), $index, $enc)

# --- Changelog.md (mirror versions.md) ---
$ver = [IO.File]::ReadAllText((Join-Path $repo 'versions.md')).TrimEnd()
$changelog = "---`nproject: PartnerTool`ntags: [partnertool, changelog]`n---`n`n# Changelog`n`n" +
             "> Mirror of ``versions.md`` in the repo (newest-first). Auto-generated by ``tools\sync-vault.ps1``.`n`n" +
             $ver + "`n"
[IO.File]::WriteAllText((Join-Path $proj 'Changelog.md'), $changelog, $enc)

# --- orphans (Code notes whose source is gone) ---
$orphans = Get-ChildItem $code -File -Filter *.md |
  Where-Object { $_.Name -ne '_Code Index.md' -and -not $expected.ContainsKey($_.Name) }
foreach ($o in $orphans) {
  if ($PruneOrphans) {
    Remove-Item -LiteralPath $o.FullName -Force
    Write-Host "sync-vault: pruned orphan (source gone): $($o.Name)"
  } else {
    Write-Host "sync-vault: ORPHAN (source gone, run -PruneOrphans to delete): $($o.Name)"
  }
}

Write-Host "sync-vault: wrote $($rows.Count) code notes + _Code Index + Changelog to '$code'."
```
